5부 구성: 1부 REST API → 2부 DB 조회 → 3부 파일 읽기 → 4부 MCP 서버 → 5부 MCP 클라이언트. 핵심 원칙 하나 — 도구는 예외로 죽지 않고 {"error": ...}로 실패를 모델에 돌려준다.

개념 ① / 실전 연동은 "실패한다"는 전제에서 시작

순수 함수는 항상 성공했다. 실제는 다르다

REST API

느려지거나(타임아웃), 5xx로 죽거나, rate limit(429)에 걸린다.

DB

커넥션이 끊기고, 잘못된 쿼리는 인젝션 위험을 낳는다.

파일

경로 조작으로 /etc/passwd를 읽으려는 path traversal이 온다.

Agent에서는 인자를 LLM이 생성하므로 위험이 더 크다. 모델이 만든 인자를 절대 신뢰하지 말고 시스템 경계에서 검증한다.

개념 ② / 신뢰성 가드레일 4종

실패를 관찰 가능한 결과로 바꾼다

가드레일목적구현
타임아웃느린 외부 호출이 전체를 멈추지 않게httpx.Client(timeout=...), DB statement_timeout
재시도일시적 오류(네트워크·429·5xx) 회복tenacity 지수 백오프
입력검증LLM이 만든 인자 방어허용 목록(allowlist), 파라미터 바인딩, 경로 정규화
에러 처리실패를 모델에게 구조화해 전달 → 회복 유도{"error": "..."} 형태로 반환(예외로 죽지 않음)

핵심 원칙: 도구는 예외를 던져 프로세스를 죽이지 않고, 실패를 결과로 모델에 돌려준다. 그래야 agent가 다른 도구로 우회하거나 사용자에게 상황을 설명할 수 있다.

개념 ④ / MCP 서버가 노출하는 3가지 + 전송

Tools · Resources · Prompts, 이번 주는 Tools

실행 가능한 함수. 이번 주 집중 — 환율·DB·파일 도구를 노출.

real_tools.py — 환율 API (타임아웃 + 재시도)

agent/real_tools.py — 재시도 + 실제 호출
import httpx
from tenacity import (retry, retry_if_exception_type,
    stop_after_attempt, wait_exponential)

HTTP_TIMEOUT_SECONDS = 5.0
MAX_RETRIES = 3
ALLOWED_CURRENCIES = {"USD","EUR","KRW","JPY","GBP","CNY"}


@retry(
    retry=retry_if_exception_type(httpx.HTTPError),
    stop=stop_after_attempt(MAX_RETRIES),
    wait=wait_exponential(multiplier=0.5, max=4),
    reraise=True,
)
def _fetch_rate(base: str, target: str) -> float:
    url = "https://api.frankfurter.app/latest"
    with httpx.Client(timeout=HTTP_TIMEOUT_SECONDS) as c:
        resp = c.get(url, params={"from": base, "to": target})
        resp.raise_for_status()   # 4xx/5xx → 재시도
        data = resp.json()
    return float(data["rates"][target])
agent/real_tools.py — 입력검증 + 에러 반환
def get_exchange_rate(base: str, target: str) -> dict:
    """base 1단위의 target 환율. 예: 'USD','KRW'."""
    base, target = base.upper(), target.upper()
    # 입력검증: 허용된 통화만 (LLM 인자 방어)
    if base not in ALLOWED_CURRENCIES \
       or target not in ALLOWED_CURRENCIES:
        return {"error": "지원하지 않는 통화입니다."}
    try:
        rate = _fetch_rate(base, target)
        return {"base": base, "target": target,
                "rate": rate}
    except Exception as exc:  # 재시도 소진·타임아웃
        return {"error": f"환율 조회 실패: {exc}"}
키가 필요 없는 공개 API(frankfurter.app). 왜 에러를 dict로? 예외를 그대로 던지면 agent 루프가 죽는다. {"error": ...}면 모델이 실패를 설명하거나 다른 방법을 시도할 수 있다.

import psycopg

DB_DSN = os.getenv("DOCPILOT_DB_DSN", "postgresql://...")
DB_TIMEOUT_MS = 3000
MAX_DB_ROWS = 20


def search_documents_db(keyword: str, limit: int = 5) -> dict:
    """kb_documents에서 keyword가 포함된 행 검색."""
    keyword = (keyword or "").strip()
    if not keyword:
        return {"error": "keyword가 비어 있습니다."}
    limit = max(1, min(int(limit), MAX_DB_ROWS))  # clamp
    try:
        with psycopg.connect(DB_DSN, connect_timeout=3,
            options=f"-c statement_timeout={DB_TIMEOUT_MS}") as conn:
            with conn.cursor() as cur:
agent/real_tools.py — %s 바인딩 (f-string 금지!)
                cur.execute(
                    """
                    SELECT id, title, content
                    FROM kb_documents
                    WHERE title ILIKE %(kw)s
                       OR content ILIKE %(kw)s
                    ORDER BY id LIMIT %(lim)s
                    """,
                    {"kw": f"%{keyword}%", "lim": limit},
                )
                rows = cur.fetchall()
        return {"count": len(rows),
                "rows": [{"id": r[0], "title": r[1],

파일 읽기 — sandbox 안에서만 (path traversal 방어)

agent/real_tools.py (이어서 추가)
from pathlib import Path

DATA_DIR = Path(os.getenv("DOCPILOT_DATA_DIR",
                          "./data")).resolve()
MAX_FILE_BYTES = 100_000


def read_file(relative_path: str) -> dict:
    """DATA_DIR 내부 텍스트 파일만 읽는다."""
    try:
        # 정규화 후 sandbox 루트 안인지 검증 (../ 차단)
        target = (DATA_DIR / relative_path).resolve()
        if not target.is_relative_to(DATA_DIR):
            return {"error": "허용된 디렉터리 밖의 경로입니다."}
        if not target.is_file():
            return {"error": f"파일이 없습니다: {relative_path}"}
        if target.stat().st_size > MAX_FILE_BYTES:
            return {"error": "파일이 너무 큽니다(100KB 초과)."}
        return {"path": relative_path,
                "content": target.read_text(encoding="utf-8")}
    except Exception as exc:
        return {"error": f"파일 읽기 실패: {exc}"}

4부 · 6단계 · 핵심

FastMCP — @mcp.tool() 하나로 도구 노출

mcp_server/server.py — stdio transport
from mcp.server.fastmcp import FastMCP
from agent.real_tools import (get_exchange_rate,
    read_file, search_documents_db)

mcp = FastMCP("docpilot-tools")


@mcp.tool()
def exchange_rate(base: str, target: str) -> dict:
    """두 통화 간 최신 환율을 조회한다."""
    return get_exchange_rate(base, target)


@mcp.tool()
def documents_search(keyword: str, limit: int = 5) -> dict:
    """kb_documents를 키워드로 검색한다."""
    return search_documents_db(keyword, limit)


@mcp.tool()
def data_file(relative_path: str) -> dict:
    """데이터 디렉터리 안의 텍스트 파일을 읽는다."""
    return read_file(relative_path)


if __name__ == "__main__":
    mcp.run(transport="stdio")
자동 생성의 힘

@mcp.tool() 하나로 도구의 이름·설명(docstring)·파라미터 스키마(타입 힌트)가 자동 생성된다. week09에서 손으로 쓰던 JSON Schema를 SDK가 대신 만든다.

클라이언트 — stdio로 붙어 나열·호출

mcp_client_demo.py
import asyncio
from mcp import ClientSession, StdioServerParameters
from mcp.client.stdio import stdio_client

# 서버를 어떻게 띄울지 지정 (파이썬 모듈)
SERVER = StdioServerParameters(command="python",
    args=["-m", "mcp_server.server"])


async def main() -> None:
    async with stdio_client(SERVER) as (read, write):
        async with ClientSession(read, write) as session:
            await session.initialize()
            # (1) 도구 목록 — 표준 프로토콜로 발견
            tools = await session.list_tools()
            for t in tools.tools:
                print(f"- {t.name}: {t.description}")
            # (2) 도구 호출
            r = await session.call_tool(
                "documents_search", {"keyword": "요금제"})
            print(r.content[0].text)


asyncio.run(main())
