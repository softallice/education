개념 ① / 왜 RAG인가

그냥 물으면 지어낸다

LLM은 학습 시점 지식만 알고 우리 회사·수업 문서는 모른다 → 그냥 물으면 할루시네이션. 문서를 통째로 프롬프트에 넣으면 컨텍스트 한도 초과 + 비용 폭발. 그래서 관련 조각만 찾아 넣는다.

RAG(Retrieval-Augmented Generation) = "질문과 관련된 문서 조각만 찾아서(Retrieval) 프롬프트에 넣고(Augment) 답을 만든다(Generation)."

개념 ③④ / 청킹 · 벡터DB

자르는 단위와 담는 그릇

청킹(chunking)

문서를 검색 단위로 자르는 것. 너무 크면 관련 없는 내용까지 딸려오고, 너무 작으면 맥락이 끊긴다.

실무 기본값: 문자 300~800자, 겹침(overlap) 10~20%. 겹침은 경계에서 문장이 잘려 뜻이 사라지는 걸 막는다.

임베딩 벡터를 저장하고 "가장 가까운 k개"를 빠르게 찾아주는 DB.

app/chunking.py — 겹침 청킹

app/chunking.py
from __future__ import annotations


def chunk_text(text: str, chunk_size: int = 600,
               overlap: int = 100) -> list[str]:
    """문자 기준 슬라이딩 윈도로 자른다(겹침 포함)."""
    text = " ".join(text.split())  # 공백/개행 정규화
    if not text:
        return []
    if chunk_size <= overlap:
        raise ValueError("chunk_size must be greater than overlap")

    chunks: list[str] = []
    start = 0
    step = chunk_size - overlap
    while start < len(text):
        end = start + chunk_size
        chunks.append(text[start:end])
        start += step
    return chunks

app/vectorstore.py — 저장 · 검색

app/vectorstore.py — 컬렉션 + upsert
from functools import lru_cache
import chromadb
from app.config import get_settings
from app.embeddings import embed   # week07 재사용


@lru_cache
def get_collection():
    """디스크에 저장되는 Chroma 컬렉션(1회 초기화)."""
    settings = get_settings()
    client = chromadb.PersistentClient(path=settings.chroma_dir)
    return client.get_or_create_collection(
        name=settings.rag_collection,
        metadata={"hnsw:space": "cosine"},  # 코사인 거리
    )


def add_chunks(doc_id: str, chunks: list[str], source: str) -> int:
    """조각들을 임베딩해 컬렉션에 upsert. 저장 개수 반환."""
    if not chunks:
        return 0
    vectors = embed(chunks)  # (n, dim), 정규화됨
    ids = [f"{doc_id}:{i}" for i in range(len(chunks))]
    metadatas = [{"doc_id": doc_id, "source": source,
                  "chunk_index": i} for i in range(len(chunks))]
    get_collection().upsert(
        ids=ids, documents=chunks,
        embeddings=vectors.tolist(), metadatas=metadatas)
    return len(chunks)
핵심 설계

임베딩은 우리가 직접 계산해 넣으므로 embedding_function은 지정하지 않는다. 컬렉션은 @lru_cache로 1회만 연다.

metadata={"hnsw:space": "cosine"}로 코사인 거리 인덱스를 쓴다. 인덱싱·질의 임베딩 모델 일치가 검색 품질의 전제.
2부 · 3단계 · 표준 검색 인터페이스

search — week09~11이 재사용할 함수

app/vectorstore.py — 공개 검색 함수
# 앵커: 이 search를 week09~11에서 그대로 "도구(tool)"로 재사용.
#   재사용은 항상 `from app.vectorstore import search`로 통일하며,
#   반환형 list[dict](keys: text, source, score)가 표준 인터페이스.
def search(query: str, top_k: int | None = None) -> list[dict]:
    """질문을 임베딩해 가장 가까운 조각 top_k를 반환."""
    settings = get_settings()
    k = top_k or settings.top_k
    q_vec = embed([query])[0]
    res = get_collection().query(
        query_embeddings=[q_vec.tolist()], n_results=k)

    hits: list[dict] = []
    docs = res.get("documents", [[]])[0]
    metas = res.get("metadatas", [[]])[0]
    dists = res.get("distances", [[]])[0]
    for doc, meta, dist in zip(docs, metas, dists):
        hits.append({
            "text": doc,
            "source": meta.get("source"),
            "chunk_index": meta.get("chunk_index"),
            "score": round(1 - dist, 4),  # 거리→유사도 근사
        })
    return hits

문서 로더 + 인덱싱 CLI

app/loader.py — txt·md·pdf → 텍스트
from __future__ import annotations
from pathlib import Path
from pypdf import PdfReader


def load_text(path: str | Path) -> str:
    path = Path(path)
    suffix = path.suffix.lower()
    if suffix in {".txt", ".md"}:
        return path.read_text(encoding="utf-8",
                              errors="ignore")
    if suffix == ".pdf":
        reader = PdfReader(str(path))
        return "\n".join(
            (page.extract_text() or "")
            for page in reader.pages)
    raise ValueError(f"unsupported file type: {suffix}")
scripts/index_docs.py — 사전 1회 인덱싱
import argparse, sys
from pathlib import Path
sys.path.insert(0, str(Path(__file__).resolve().parent.parent))

from app.config import get_settings
from app.chunking import chunk_text
from app.loader import load_text
from app.vectorstore import add_chunks


def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("path", help="파일 또는 폴더")
    args = parser.parse_args()
    s = get_settings(); target = Path(args.path)
    files = [target] if target.is_file() else sorted(
        p for p in target.rglob("*")
        if p.suffix.lower() in {".txt", ".md", ".pdf"})
    total = 0
    for f in files:
        chunks = chunk_text(load_text(f), s.chunk_size, s.chunk_overlap)
        n = add_chunks(doc_id=f.stem, chunks=chunks, source=f.name)
        total += n; print(f"indexed {f.name}: {n} chunks")
    print(f"done. total chunks = {total}")

app/rag.py — 검색 → 주입 → 생성

app/rag.py
from app import llm                       # week06
from app.vectorstore import search        # 표준 검색

RAG_SYSTEM = (
    "You are docpilot, a document-grounded assistant. "
    "Answer ONLY using the provided context. "
    "If the answer is not in the context, say "
    "'문서에서 근거를 찾지 못했습니다.' in Korean. "
    "Cite sources by their [source] tag. Answer in Korean."
)


def build_context(hits: list[dict]) -> str:
    blocks = [f"[source: {h['source']} #{h['chunk_index']}]\n{h['text']}"
              for h in hits]
    return "\n\n---\n\n".join(blocks)


async def answer(question: str, top_k=None) -> dict:
    hits = search(question, top_k=top_k)
    if not hits:
        return {"answer": "인덱싱된 문서가 없습니다...", "sources": []}
    context = build_context(hits)
    user_prompt = (f"컨텍스트:\n{context}\n\n질문: {question}\n\n"
                   "위 컨텍스트만 근거로 답하고, 출처를 밝혀줘.")
    messages = llm.build_messages(user_prompt, system=RAG_SYSTEM)
    reply = await llm.complete(messages, temperature=0.1)
    sources = [{"source": h["source"], "chunk_index": h["chunk_index"],
                "score": h["score"]} for h in hits]
    return {"answer": reply, "sources": sources}
할루시네이션 억제

"컨텍스트에 없으면 모른다고 답하라"는 규칙을 시스템 프롬프트에 넣는다. 사실 답변이라 temperature=0.1로 낮게.

각 조각을 [source: 파일 #인덱스] 태그와 함께 넣어 출처 인용이 가능하게 한다.

from pydantic import BaseModel, Field
from app import rag


class AskRequest(BaseModel):
    question: str = Field(..., min_length=1,
                          max_length=2000)
    top_k: int | None = Field(
        default=None, ge=1, le=20)


@app.post("/ask")
async def ask(req: AskRequest) -> dict:
    """문서 기반(RAG) 질의응답. 답변 + 출처."""
    return await rag.answer(req.question, top_k=req.top_k)

def search(query: str, top_k=None) -> list[dict]:
    k = top_k or get_settings().top_k
    q_vec = embed([query])[0]
    with _connect() as conn, conn.cursor() as cur:
        # <=> 는 코사인 거리 연산자. 1 - 거리 = 유사도
        cur.execute(
            "SELECT source, chunk_index, content, "
            "1 - (embedding <=> %s) AS score "
            "FROM doc_chunks ORDER BY embedding <=> %s "
            "LIMIT %s", (q_vec, q_vec, k))
        rows = cur.fetchall()
    return [{"source": r[0], "chunk_index": r[1],
             "text": r[2], "score": round(float(r[3]), 4)}
            for r in rows]
